CREATE TABLE  "FIREFORESTDATA" 
   (	"DATETIME" TIMESTAMP (6) NOT NULL ENABLE, 
	"SENSOR_ID" VARCHAR2(4000) NOT NULL ENABLE, 
	"LOCATION" VARCHAR2(4000) NOT NULL ENABLE, 
	"HUMIDITY_PERCENT" FLOAT(126) NOT NULL ENABLE, 
	"TEMPERATURE_DEGREES" FLOAT(126) NOT NULL ENABLE, 
	"C0DANGER" NUMBER NOT NULL ENABLE, 
	 CONSTRAINT "FIREFORESTDATA_PK" PRIMARY KEY ("SENSOR_ID") ENABLE
   ) ;

CREATE OR REPLACE TRIGGER  "BI_FIREFORESTDATA" 
  before insert on "FIREFORESTDATA"               
  for each row  
begin   
  if :NEW."SENSOR_ID" is null then 
    select "FIREFORESTDATA_SEQ".nextval into :NEW."SENSOR_ID" from dual; 
  end if; 
end; 

/
ALTER TRIGGER  "BI_FIREFORESTDATA" ENABLE;
